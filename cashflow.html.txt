const HOLIDAYS_2026 = new Set([
  '2026-01-01','2026-01-12','2026-02-11','2026-02-23','2026-03-20',
  '2026-04-29','2026-05-03','2026-05-04','2026-05-05','2026-05-06',
  '2026-07-20','2026-08-11','2026-09-21','2026-09-22','2026-09-23',
  '2026-10-12','2026-11-03','2026-11-23','2026-12-23',
]);
const HOLIDAYS_2027 = new Set([
  '2027-01-01','2027-01-11','2027-02-11','2027-02-23','2027-03-22',
  '2027-04-29','2027-05-03','2027-05-04','2027-05-05',
  '2027-07-19','2027-08-11','2027-09-20','2027-09-23',
  '2027-10-11','2027-11-03','2027-11-23',
]);

function isHoliday(date) {
  const y = date.getFullYear();
  const str = date.toISOString().slice(0,10);
  if (y === 2026) return HOLIDAYS_2026.has(str);
  if (y === 2027) return HOLIDAYS_2027.has(str);
  return false;
}

// 翌営業日を返す（土日・祝日をスキップ）
function nextBusinessDay(date) {
  const d = new Date(date);
  while (d.getDay() === 0 || d.getDay() === 6 || isHoliday(d)) {
    d.setDate(d.getDate() + 1);
  }
  return d;
}

// 月末日を返す
function lastDayOfMonth(year, month) {
  return new Date(year, month + 1, 0);
}

// m=0-11, 表示用の引落日文字列を返す
function calcDebitDate(item, m) {
  const year = 2026; // 令和8年度
  let base;
  if (item.debitRule === 'nextMonth10') {
    // 翌月10日
    const nm = (m + 1) % 12;
    const ny = m === 11 ? year + 1 : year;
    base = new Date(ny, nm, 10);
  } else if (item.debitRule === 'same27') {
    // 同月27日
    base = new Date(year, m, 27);
  } else if (item.debitRule === 'monthEnd') {
    // 月末日
    base = lastDayOfMonth(year, m);
  } else {
    return '';
  }
  const bd = nextBusinessDay(base);
  return (bd.getMonth() + 1) + '/' + bd.getDate();
}

function saveActualTransfer(key, val) {
  const num = parseComma(val);
  if (num > 0) {
    localStorage.setItem(key, String(num));
  } else {
    localStorage.removeItem(key);
  }
}

// ── Export / Import ──
function exportData(fromSave) {
  const data = {};
  for (let i = 0; i < localStorage.length; i++) {
    const k = localStorage.key(i);
    if (k && k.startsWith('cf8_')) {
      data[k] = localStorage.getItem(k);
    }
  }
  const now = new Date();
  const ts = now.getFullYear() + '/' +
    String(now.getMonth()+1).padStart(2,'0') + '/' +
    String(now.getDate()).padStart(2,'0') + ' ' +
    String(now.getHours()).padStart(2,'0') + ':' +
    String(now.getMinutes()).padStart(2,'0');
  data['_exportedAt'] = ts;
  const json = JSON.stringify(data);

  // テキストエリアに出力
  const ta = document.getElementById('importText');
  ta.value = json;

  // バックアップ案内エリアを表示
  const backupArea = document.getElementById('backupGuide');
  if (backupArea) backupArea.style.display = 'block';

  if (!fromSave) {
    showSyncMsg('✓ データを出力しました。下の手順でメモアプリに保存してください。', 'ok');
  }
}

function openNoteApp() {
  // テキストエリアの内容を取得
  const ta = document.getElementById('importText');
  const text = ta.value;
  if (!text) { exportData(); return; }
  // iOSメモアプリをURLスキームで開く（テキストをクリップボード経由で）
  try {
    if (navigator.clipboard && navigator.clipboard.writeText) {
      navigator.clipboard.writeText(text).then(() => {
        showSyncMsg('✓ クリップボードにコピーしました！メモアプリを開いて貼り付けてください。', 'ok');
        // メモアプリを開く
        setTimeout(() => { window.location.href = 'mobilenotes://'; }, 800);
      }).catch(() => {
        showSyncMsg('下のテキストを長押し→全選択→コピーしてメモアプリに貼り付けてください。', 'err');
      });
    } else {
      showSyncMsg('下のテキストを長押し→全選択→コピーしてメモアプリに貼り付けてください。', 'ok');
    }
  } catch(e) {
    showSyncMsg('下のテキストを長押し→全選択→コピーしてメモアプリに貼り付けてください。', 'ok');
  }
}

function importData() {
  const text = document.getElementById('importText').value.trim();
  if (!text) {
    showSyncMsg('データが入力されていません。', 'err');
    return;
  }
  try {
    const data = JSON.parse(text);
    let count = 0;
    Object.keys(data).forEach(k => {
      if (k.startsWith('cf8_')) {
        localStorage.setItem(k, data[k]);
        count++;
      }
    });
    document.getElementById('importText').value = '';
    showSyncMsg(`✓ ${count}件のデータを取り込みました。ページを再読み込みします…`, 'ok');
    setTimeout(() => location.reload(), 1500);
  } catch(e) {
    showSyncMsg('データの形式が正しくありません。エクスポートしたデータをそのまま貼り付けてください。', 'err');
  }
}

}

function getDynamicRows(prefix) {
  const n = parseInt(lsGet(currentMonth, prefix+'_count') || '1');
  return Math.max(1, n);
}

function renderDynamicRows(sec, prefix) {
  const count = getDynamicRows(prefix);
  for (let i = 1; i <= count; i++) {
    sec.appendChild(makeDynamicRow(prefix, i));
  }
  // ＋追加ボタン
  const addBtn = document.createElement('button');
  addBtn.className = 't-add-btn';
  addBtn.style.marginLeft = '10px';
  addBtn.textContent = '＋ 行を追加';
  addBtn.onclick = () => {
    const cur = getDynamicRows(prefix);
    lsSet(currentMonth, prefix+'_count', String(cur + 1));
    renderExpense();
    recalc();
    initCommaInputs();
  };
  sec.appendChild(addBtn);
}

function saveDynamic(prefix, i) {
  const id = prefix + i;
  const nm = document.getElementById('nm_'+id);
  const bk = document.getElementById('bk_'+id);
  const ex = document.getElementById('ex_'+id);
  if (nm) lsSet(currentMonth, id+'_name', nm.value);
  if (bk) lsSet(currentMonth, id+'_bank', bk.value);
  if (ex) lsSet(currentMonth, id, String(parseComma(ex.value)));
  recalc();
}

function deleteDynamicRow(prefix, i) {
  // 該当行のデータを削除し、以降の行を詰める
  const count = getDynamicRows(prefix);
  for (let j = i; j < count; j++) {
    const cur = prefix + j;
    const next = prefix + (j+1);
    lsSet(currentMonth, cur+'_name', lsGet(currentMonth, next+'_name'));
    lsSet(currentMonth, cur+'_bank', lsGet(currentMonth, next+'_bank'));
    lsSet(currentMonth, cur,         lsGet(currentMonth, next));
    lsSet(currentMonth, cur+'_paid', lsGet(currentMonth, next+'_paid'));
  }
  // 最後の行をクリア
  const last = prefix + count;
  lsSet(currentMonth, last+'_name', '');
  lsSet(currentMonth, last+'_bank', '');
  lsSet(currentMonth, last, '');
  lsSet(currentMonth, last+'_paid', '');
  // カウントを減らす（最低1行）
  if (count > 1) lsSet(currentMonth, prefix+'_count', String(count - 1));
  renderExpense();
  recalc();
  initCommaInputs();
}

// ── Recalc ──
function fmt(n) { return '¥' + Math.round(n).toLocaleString('ja-JP'); }
function gn(id) { const e=document.getElementById(id); return e?parseComma(e.value):0; }

function recalc() {
  // 支出合計（支払済みはスキップ）
  let totalEx = 0;
  EXPENSES.forEach(item => {
    const hasOnly = !!item.onlyMonths;
    const active  = !hasOnly || item.onlyMonths.includes(currentMonth);
    if (!active) return;
    const paid = lsGet(currentMonth, item.id+'_paid') === '1';
    if (paid) return;
    const deleted2 = JSON.parse(localStorage.getItem(`cf8_m${currentMonth}_deleted`) || '[]');
    if (deleted2.includes(item.id)) return;
    if (item.fixed !== undefined) { totalEx += item.fixed; return; }
    totalEx += gn('ex_'+item.id);
  });
  // 動的行（臨時・固定）の合計
  ['card_extra','temp','custom'].forEach(prefix => {
    const count = parseInt(lsGet(currentMonth, prefix+'_count') || '1');
    for (let i = 1; i <= count; i++) {
      const id = prefix + i;
      const paid = lsGet(currentMonth, id+'_paid') === '1';
      if (paid) continue;
      totalEx += parseComma(lsGet(currentMonth, id) || '0');
    }
  });

  // bankHint付き支出を口座別に集計
  const bankAutoDeduct = {};
  window._bankAutoDeduct = bankAutoDeduct;
  EXPENSES.forEach(item => {
    const hasOnly = !!item.onlyMonths;
    const active  = !hasOnly || item.onlyMonths.includes(currentMonth);
    if (!active) return;
    // 支払済み・非表示はスキップ
    const paid = lsGet(currentMonth, item.id+'_paid') === '1';
    if (paid) return;
    if (item.deletable && lsGet(currentMonth, item.id+'_hidden') === '1') return;
    // 通常のbankHint
    const deletedB = JSON.parse(localStorage.getItem(`cf8_m${currentMonth}_deleted`) || '[]');
    if (deletedB.includes(item.id)) return;
    if (item.bankHint) {
      const v = item.fixed !== undefined ? item.fixed : gn('ex_'+item.id);
      bankAutoDeduct[item.bankHint] = (bankAutoDeduct[item.bankHint] || 0) + v;
    }
  });
  // 動的行の口座別集計
  ['card_extra','temp','custom'].forEach(prefix => {
    const count = parseInt(lsGet(currentMonth, prefix+'_count') || '1');
    for (let i = 1; i <= count; i++) {
      const id = prefix + i;
      const paid = lsGet(currentMonth, id+'_paid') === '1';
      if (paid) continue;
      const bankId = lsGet(currentMonth, id+'_bank');
      if (!bankId) continue;
      const v = parseComma(lsGet(currentMonth, id) || '0');
      if (v > 0) bankAutoDeduct[bankId] = (bankAutoDeduct[bankId] || 0) + v;
    }
  });

  // 資金移動の実績額を口座ごとに集計
  const transferIn  = {};  // 受取口座への入金
  const transferOut = {};  // 送出口座からの出金
  // LocalStorageからtransfer_actualキーを走査
  for (let i = 0; i < localStorage.length; i++) {
    const k = localStorage.key(i);
    if (!k || !k.startsWith(`cf8_m${currentMonth}_transfer_actual_`)) continue;
    const parts = k.replace(`cf8_m${currentMonth}_transfer_actual_`, '').split('_');
    if (parts.length < 2) continue;
    const fromId = parts[0];
    const toId   = parts[1];
    const amt    = parseComma(localStorage.getItem(k) || '0');
    if (amt <= 0) continue;
    transferOut[fromId] = (transferOut[fromId] || 0) + amt;
    transferIn[toId]    = (transferIn[toId]    || 0) + amt;
  }

  // 口座
  let bankStart = 0, bankEnd = 0;
  BANKS.forEach(b => {
    const s = gn('bs_'+b.id);
    const d = gn('bd_'+b.id);
    const autoD = bankAutoDeduct[b.id] || 0;
    const tOut  = transferOut[b.id] || 0;
    const tIn   = transferIn[b.id]  || 0;
    const e = s - d - autoD - tOut + tIn;
    bankStart += s;
    bankEnd   += e;
    const el = document.getElementById('be_'+b.id);
    if (el) {
      el.textContent = fmt(e);
      el.className = 'bg-val ' + (e >= 0 ? 'positive' : 'negative');
    }
    const autoEl = document.getElementById('bauto_'+b.id);
    if (autoEl) {
      autoEl.textContent = autoD > 0 ? 'カード自動引落 −' + fmt(autoD) : '';
    }
  });

  // Summary
  document.getElementById('sumExpense').textContent   = fmt(totalEx);
  document.getElementById('sumBankStart').textContent = fmt(bankStart);
  const endEl = document.getElementById('sumBankEnd');
  endEl.textContent = fmt(bankEnd);
  endEl.className   = 'val ' + (bankEnd >= 0 ? 'col-plus' : 'col-minus');

  const te = document.getElementById('totalExpense');
  if (te) te.textContent = fmt(totalEx);
  const tbe = document.getElementById('totalBankEnd');
  if (tbe) tbe.textContent = fmt(bankEnd);

  // Transfer suggestions
  renderTransfer();
  // Re-attach comma formatting to any new inputs
  initCommaInputs();

  // 口座未選択チェック（動的行）
  const unassigned = [];
  ['card_extra','temp','custom'].forEach(prefix => {
    const count = parseInt(lsGet(currentMonth, prefix+'_count') || '1');
    for (let i = 1; i <= count; i++) {
      const id = prefix + i;
      const paid = lsGet(currentMonth, id+'_paid') === '1';
      if (paid) continue;
      const amt    = parseComma(lsGet(currentMonth, id) || '0');
      const bankId = lsGet(currentMonth, id+'_bank');
      const name   = lsGet(currentMonth, id+'_name') || '（名称未入力）';
      const bk = document.getElementById('bk_'+id);
      if (amt > 0 && !bankId) {
        unassigned.push(name);
        if (bk) bk.style.border = '2px solid var(--red)';
      } else {
        if (bk) bk.style.border = '';
      }
    }
  });

  // Alert
  const al = document.getElementById('cfAlert');
  al.className = 'alert';
  if (unassigned.length > 0) {
    al.className = 'alert danger';
    al.innerHTML = `⚠️ 口座未選択の支出があります：<b>${unassigned.join('・')}</b><br><span style="font-size:11px;font-weight:400;">口座を選択しないと残高計算に反映されません。</span>`;
  } else if (bankEnd < 0) {
    al.className = 'alert danger';
    al.innerHTML = '⚠️ 月末残高がマイナスになる見込みです。';
  } else if (bankEnd < 1000000) {
    al.className = 'alert warning';
    al.innerHTML = '⚡ 月末残高が100万円を下回る見込みです。';
  } else {
    al.className = 'alert ok';
    al.innerHTML = '✓ 残高は良好です。';
  }
}

// ── Render Transfer Plan ──
function renderTransfer() {
  const sec = document.getElementById('transferSection');
  if (!sec) return;

  // Collect bank balances: start, deductions (fixed card/tax), manual deductions, end
  const bankData = {};
  BANKS.forEach(b => {
    const s = gn('bs_'+b.id);
    const d = gn('bd_'+b.id);
    const autoD = (window._bankAutoDeduct || {})[b.id] || 0;
    bankData[b.id] = { id: b.id, label: b.label, color: b.color, start: s, manual: d, auto: autoD, end: s - d - autoD };
  });

  // Find banks with shortfall (end < 0 or end < threshold)
  const THRESHOLD = 0; // 残高マイナスのみ不足と判定
  const shortfalls = [];
  const surpluses  = [];

  BANKS.forEach(b => {
    const d = bankData[b.id];
    if (d.end < THRESHOLD) {
      shortfalls.push({ ...d, id: b.id, need: THRESHOLD - d.end });
    } else {
      surpluses.push({ ...d, id: b.id, avail: d.end - THRESHOLD });
    }
  });

  // Build transfer plan: cover shortfalls from surpluses
  const suggestions = [];
  const surplusCopy = surpluses.map(s => ({ ...s }));

  shortfalls.forEach(sf => {
    let remaining = sf.need;
    surplusCopy.forEach(sp => {
      if (remaining <= 0 || sp.avail <= 0) return;
      const move = Math.min(remaining, sp.avail);
      suggestions.push({ from: sp, to: sf, amount: move });
      sp.avail -= move;
      remaining -= move;
    });
  });

  // Render
  sec.innerHTML = '';

  // 固定振込を一覧表示
  // 大原家生活費：三井住友が80,000円以上あれば三井住友から、なければ余剰口座から
  const smbcEnd = bankData['smbc'] ? bankData['smbc'].end : 0;
  const oharaFrom = smbcEnd >= 80000
    ? { fromColor:'#0055b0', from:'三井住友銀行' }
    : (() => {
        const surplus = Object.values(bankData).find(b => b.id !== 'rakuten' && b.id !== 'smbc' && b.end >= 80000);
        return surplus
          ? { fromColor: surplus.color, from: surplus.label }
          : { fromColor:'#c0392b', from:'⚠️ 残高不足' };
      })();

  const fixedTransfers = [
    { fromColor:'#0055b0', from:'三井住友銀行', to:'㈱マツガミ',        amount:'¥600,000', note:'リース料 月末振込' },
    { fromColor: oharaFrom.fromColor, from: oharaFrom.from, to:'楽天銀行（大原家生活費）', amount:'¥80,000', note:'大原家生活費 → 楽天銀行へ入金後振込' },
  ];

  const fixedLabel = document.createElement('div');
  fixedLabel.style.cssText = 'font-size:11px;font-weight:700;letter-spacing:.06em;color:var(--ink-faint);margin-bottom:8px;';
  fixedLabel.textContent = '振込（固定）';
  sec.appendChild(fixedLabel);

  fixedTransfers.forEach(t => {
    const div = document.createElement('div');
    div.className = 'transfer-item';
    div.style.background = '#fff8e8';
    div.style.borderColor = '#f0dfa0';
    div.innerHTML = `
      <span class="t-bank" style="display:flex;align-items:center;gap:5px;">
        <span style="display:inline-block;width:8px;height:8px;border-radius:50%;background:${t.fromColor}"></span>
        ${t.from}
      </span>
      <span class="t-arrow" style="color:var(--yellow);">→</span>
      <span class="t-bank">${t.to}</span>
      <span class="t-amount" style="color:var(--yellow);">${t.amount}</span>
      <span class="t-note">${t.note}</span>`;
    sec.appendChild(div);
  });

  // 不足口座への移動提案ラベル
  const autoLabel = document.createElement('div');
  autoLabel.style.cssText = 'font-size:11px;font-weight:700;letter-spacing:.06em;color:var(--ink-faint);margin:12px 0 8px;';
  autoLabel.textContent = '口座間移動（自動提案）';
  sec.appendChild(autoLabel);

  if (suggestions.length === 0 && shortfalls.length === 0) {
    const ok = document.createElement('div');
    ok.className = 'transfer-empty';
    ok.innerHTML = '✓ 全口座で残高が十分です。資金移動は不要です。';
    sec.appendChild(ok);
  } else if (suggestions.length === 0 && shortfalls.length > 0) {
    const ng = document.createElement('div');
    ng.style.cssText = 'color:var(--red);font-size:13px;font-weight:600;padding:8px 0;';
    ng.innerHTML = '⚠️ 不足を補える余剰口座がありません。入金・追加資金が必要です。';
    sec.appendChild(ng);
  } else {
    suggestions.forEach((s, idx) => {
      const actualKey = `cf8_m${currentMonth}_transfer_actual_${s.from.id}_${s.to.id}`;
      const actualVal = localStorage.getItem(actualKey) || '';
      const actualNum = actualVal ? parseComma(actualVal) : Math.ceil(s.amount);
      const afterBalance = Math.ceil(s.to.end + actualNum);

      const div = document.createElement('div');
      div.className = 'transfer-item';
      div.innerHTML = `
        <span class="t-bank" style="display:flex;align-items:center;gap:5px;">
          <span style="display:inline-block;width:8px;height:8px;border-radius:50%;background:${s.from.color}"></span>
          ${s.from.label}
        </span>
        <span class="t-arrow">→</span>
        <span class="t-bank" style="display:flex;align-items:center;gap:5px;">
          <span style="display:inline-block;width:8px;height:8px;border-radius:50%;background:${s.to.color}"></span>
          ${s.to.label}
        </span>
        <div style="margin-left:auto;display:flex;align-items:center;gap:6px;">
          <span style="font-size:10px;color:var(--ink-faint);">提案</span>
          <span class="t-amount" style="font-size:13px;">¥${Math.ceil(s.amount).toLocaleString('ja-JP')}</span>
        </div>
        <div style="width:100%;display:flex;align-items:center;gap:8px;margin-top:6px;padding-top:6px;border-top:1px solid #dde1ec;">
          <span style="font-size:11px;color:var(--ink-faint);white-space:nowrap;">実際の振込額</span>
          <input class="amt-input" type="text" 
            id="ta_${s.from.id}_${s.to.id}"
            value="${actualVal ? Number(actualVal).toLocaleString('ja-JP') : ''}"
            placeholder="${Math.ceil(s.amount).toLocaleString('ja-JP')}"
            style="width:120px;font-size:13px;margin-left:auto;"
            onblur="saveActualTransfer('${actualKey}', this.value); recalc();"
            onkeydown="if(event.key==='Enter'){this.blur();}">
          <span style="font-size:12px;color:var(--ink-faint);">円</span>
        </div>
        <span class="t-note">移動後の${s.to.label}残高（試算）：¥${afterBalance.toLocaleString('ja-JP')}</span>`;
      sec.appendChild(div);
    });
  }

  // Manual transfer input section
  const manual = document.createElement('div');
  manual.className = 'transfer-manual';
  const manualRows = window._manualTransfers || [{ from:'', to:'', amt:'' }];
  window._manualTransfers = manualRows;

  manual.innerHTML = '<div class="transfer-manual-lbl">手動で追加する移動</div>';
  manualRows.forEach((row, i) => {
    const bankOptions = BANKS.map(b => `<option value="${b.id}" ${row.from===b.id?'selected':''}>${b.label}</option>`).join('');
    const bankOptions2 = BANKS.map(b => `<option value="${b.id}" ${row.to===b.id?'selected':''}>${b.label}</option>`).join('');
    const r = document.createElement('div');
    r.className = 't-manual-row';
    r.innerHTML = `
      <select class="t-select" onchange="updateManual(${i},'from',this.value)">
        <option value="">移動元</option>${bankOptions}
      </select>
      <span class="t-arrow-label">→</span>
      <select class="t-select" onchange="updateManual(${i},'to',this.value)">
        <option value="">移動先</option>${bankOptions2}
      </select>
      <input class="t-manual-amt" type="number" placeholder="金額" value="${row.amt}"
        oninput="updateManual(${i},'amt',this.value)">
      <span class="unit">円</span>`;
    manual.appendChild(r);
  });

  const addBtn = document.createElement('button');
  addBtn.className = 't-add-btn';
  addBtn.textContent = '＋ 行を追加';
  addBtn.onclick = () => { window._manualTransfers.push({from:'',to:'',amt:''}); recalc(); };
  manual.appendChild(addBtn);
  sec.appendChild(manual);
}

function updateManual(i, field, val) {
  if (!window._manualTransfers) window._manualTransfers = [];
  if (!window._manualTransfers[i]) window._manualTransfers[i] = {};
  window._manualTransfers[i][field] = val;
}

// ── Comma Formatting ──
function toHalf(str) {
  // Convert full-width numbers to half-width
  return str.replace(/[０-９]/g, s => String.fromCharCode(s.charCodeAt(0) - 0xFEE0));
}

function formatComma(val) {
  const n = parseFloat(toHalf(String(val)).replace(/,/g, '')) || 0;
  if (n === 0) return '';
  return n.toLocaleString('ja-JP');
}

function parseComma(val) {
  return parseFloat(toHalf(String(val)).replace(/,/g, '')) || 0;
}

function attachCommaFormat(input) {
  // On focus: show raw number for editing
  input.addEventListener('focus', function() {
    const raw = parseComma(this.value);
    this.value = raw === 0 ? '' : String(raw);
    this.type = 'number';
  });
  // On blur: show comma-formatted
  input.addEventListener('blur', function() {
    this.type = 'text';
    const raw = parseComma(this.value);
    this.value = raw === 0 ? '' : raw.toLocaleString('ja-JP');
    autoSave();
    recalc();
  });
  // On input: convert full-width to half-width live
  input.addEventListener('input', function() {
    const pos = this.selectionStart;
    const converted = toHalf(this.value);
    if (converted !== this.value) {
      this.value = converted;
      this.setSelectionRange(pos, pos);
    }
  });
  // Init: format existing value
  const raw = parseComma(input.value);
  input.type = 'text';
  input.value = raw === 0 ? '' : raw.toLocaleString('ja-JP');
}

function initCommaInputs() {
  document.querySelectorAll('.amt-input:not([data-comma])').forEach(el => {
    el.setAttribute('data-comma', '1');
    attachCommaFormat(el);
  });
}

function toggleMonthly(btn) {
  const type   = btn.dataset.monthlyType;
  const msKey  = type === 'dynamic'
    ? btn.dataset.monthlyPrefix + btn.dataset.monthlyI + '_monthly_set'
    : btn.dataset.monthlyId + '_monthly_set';
  const isSet  = lsGet(currentMonth, msKey) === '1';

  if (isSet) {
    // 解除
    if (type === 'static') {
      const itemId = btn.dataset.monthlyId;
      for (let m = currentMonth + 1; m <= 11; m++) {
        localStorage.removeItem(lsKey(m, itemId));
      }
    } else {
      const prefix = btn.dataset.monthlyPrefix;
      const i      = btn.dataset.monthlyI;
      const id     = prefix + i;
      for (let m = currentMonth + 1; m <= 11; m++) {
        localStorage.removeItem(lsKey(m, id));
        localStorage.removeItem(lsKey(m, id+'_name'));
        localStorage.removeItem(lsKey(m, id+'_bank'));
      }
    }
    lsSet(currentMonth, msKey, '0');
    setMonthlyBtnActive(btn, false);
    showToast('毎月定額を解除しました');
    return;
  }

  // 設定
  let val = '', name = '', bank = '', count = 0;
  if (type === 'static') {
    const itemId = btn.dataset.monthlyId;
    const el = document.getElementById('ex_'+itemId);
    val = el ? String(parseComma(el.value)) : '0';
    for (let m = currentMonth + 1; m <= 11; m++) { lsSet(m, itemId, val); count++; }
  } else {
    const prefix = btn.dataset.monthlyPrefix;
    const i      = btn.dataset.monthlyI;
    const id     = prefix + i;
    const ex = document.getElementById('ex_'+id);
    const nm = document.getElementById('nm_'+id);
    const bk = document.getElementById('bk_'+id);
    val  = ex ? String(parseComma(ex.value)) : '';
    name = nm ? nm.value : '';
    bank = bk ? bk.value : '';
    for (let m = currentMonth + 1; m <= 11; m++) {
      const mCount = parseInt(localStorage.getItem(lsKey(m, prefix+'_count')) || '1');
      if (parseInt(i) > mCount) lsSet(m, prefix+'_count', String(i));
      lsSet(m, id, val);
      if (name) lsSet(m, id+'_name', name);
      if (bank) lsSet(m, id+'_bank', bank);
      count++;
    }
  }
  lsSet(currentMonth, msKey, '1');
  setMonthlyBtnActive(btn, true);
  showToast(`✓ ${count}ヶ月分にコピーしました`);
}

function autoSave() {
  // Persist inputDate on every change too
  const dateEl = document.getElementById('inputDate');
  if (dateEl && dateEl.value) lsSet(currentMonth, 'inputDate', dateEl.value);
  EXPENSES.forEach(item => {
    if (item.fixed !== undefined) return;
    const el = document.getElementById('ex_'+item.id);
    if (el && !el.readOnly) lsSet(currentMonth, item.id, String(parseComma(el.value)));
    if (item.custom) {
      const nm = document.getElementById('nm_'+item.id);
      if (nm) lsSet(currentMonth, item.id+'_name', nm.value);
      if (item.withBank) {
        const bk = document.getElementById('bk_'+item.id);
        if (bk) lsSet(currentMonth, item.id+'_bank', bk.value);
      }
    }

  });
  BANKS.forEach(b => {
    const s = document.getElementById('bs_'+b.id);
    const d = document.getElementById('bd_'+b.id);
    if (s) lsSet(currentMonth, 'bs_'+b.id, String(parseComma(s.value)));
    if (d) lsSet(currentMonth, 'bd_'+b.id, String(parseComma(d.value)));
  });
  recalc();
}

  const now = new Date();
  const ts = now.getFullYear() + '/' +
    String(now.getMonth()+1).padStart(2,'0') + '/' +
    String(now.getDate()).padStart(2,'0') + ' ' +
    String(now.getHours()).padStart(2,'0') + ':' +

  exportData(true);

  const t = document.getElementById('toast');
  t.classList.add('show');
  setTimeout(()=>t.classList.remove('show'), 2000);
}
